---
title: "Criar lead"
api: "POST /user/lead"
description: "Cria um novo lead no sistema OmniTech"
---

Este endpoint permite criar um novo lead no sistema OmniTech.

### Corpo da Requisição

<ParamField body="phone_number" type="string" required>
  Número de telefone do lead no formato E.164 (ex.: +1234567890)
</ParamField>

<ParamField body="campaign_id" type="integer" required>
  ID da campanha para a qual o lead será criado
</ParamField>

<ParamField body="variables" type="object">
  Variáveis a serem associadas ao lead
  
  <Expandable title="propriedades de variables">
    <ParamField body="customer_name" type="string" default="<string>">
      Nome do cliente
    </ParamField>
    
    <ParamField body="email" type="string" default="<string>">
      E-mail do cliente
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="allow_dupplicate" type="boolean">
  Define se leads duplicados são permitidos em uma campanha
</ParamField>

<ParamField body="secondary_contacts" type="array">
  Array de contatos secundários a serem criados junto com o lead principal
  
  <Expandable title="propriedades de secondary_contacts">
    <ParamField body="phone_number" type="string" required>
      Número de telefone do contato secundário no formato E.164
    </ParamField>
    
    <ParamField body="variables" type="object">
      Variáveis a serem associadas ao contato secundário
      
      <Expandable title="propriedades de variables">
        <ParamField body="customer_name" type="string" default="<string>">
          Nome do contato secundário
        </ParamField>
        
        <ParamField body="email" type="string" default="<string>">
          E-mail do contato secundário
        </ParamField>
      </Expandable>
    </ParamField>
  </Expandable>
</ParamField>

### Resposta

<ResponseField name="message" type="string" default="Lead created successfully">
  Mensagem da resposta
</ResponseField>

<ResponseField name="data" type="object">
  Objeto de dados do lead
  
  <Expandable title="propriedades de data">
    <ResponseField name="id" type="integer">
      ID único do lead criado
    </ResponseField>
    
    <ResponseField name="campaign_id" type="integer">
      ID da campanha à qual este lead pertence
    </ResponseField>

    <ResponseField name="phone_number" type="string">
      Número de telefone do lead no formato E.164
    </ResponseField>
    
    <ResponseField name="variables" type="object">
      Variáveis associadas ao lead
      
      <Expandable title="propriedades de variables">
        <ResponseField name="customer_name" type="string" default="<string>">
          Nome do cliente
        </ResponseField>
        
        <ResponseField name="email" type="string" default="<string>">
          E-mail do cliente
        </ResponseField>
      </Expandable>
    </ResponseField>
    
    <ResponseField name="status" type="string" default="created">
      Status do lead
    </ResponseField>
    
    <ResponseField name="created_at" type="string">
      Data e hora de criação do lead
    </ResponseField>
    
    <ResponseField name="updated_at" type="string">
      Data e hora da última atualização do lead
    </ResponseField>

    <ResponseField name="campaign" type="object">
      Informações da campanha
      <Expandable title="propriedades de campaign">
        <ResponseField name="id" type="integer">
          ID da campanha
        </ResponseField>
        <ResponseField name="name" type="string">
          Nome da campanha
        </ResponseField>
      </Expandable>
    </ResponseField>

    <ResponseField name="secondary_contacts" type="array">
      Array de contatos secundários associados a este lead
      <Expandable title="propriedades de secondary_contacts">
        <ResponseField name="id" type="integer">
          ID do contato secundário
        </ResponseField>
        <ResponseField name="phone_number" type="string">
          Número de telefone do contato secundário no formato E.164
        </ResponseField>
        <ResponseField name="variables" type="object">
          Variáveis associadas ao contato secundário
        </ResponseField>
        <ResponseField name="status" type="string">
          Status do contato secundário
        </ResponseField>
        <ResponseField name="created_at" type="string">
          Data e hora de criação do contato secundário
        </ResponseField>
        <ResponseField name="updated_at" type="string">
          Data e hora da última atualização do contato secundário
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>
