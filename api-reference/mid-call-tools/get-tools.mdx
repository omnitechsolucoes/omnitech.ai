---
title: "Listar ferramentas de uso durante a chamada"
api: "GET /user/tools"
description: "Recupera todas as ferramentas de uso durante a chamada"
---

Este endpoint permite recuperar todas as ferramentas de uso durante a chamada. As ferramentas de mid call permitem que seus assistentes de IA interajam com APIs externas durante uma ligação.

### Headers

<ParamField header="Authorization" type="string" required>
  Token Bearer para autenticação
</ParamField>

<ParamField header="Content-Type" type="string" required>
  Deve ser `application/json`
</ParamField>

<ParamField header="Accept" type="string" required>
  Deve ser `application/json`
</ParamField>

### Campos da Resposta

<ResponseField name="data" type="array">
  Array de ferramentas de uso durante a chamada
  <Expandable title="propriedades">
    <ResponseField name="id" type="integer">
      Identificador único da ferramenta
    </ResponseField>

    <ResponseField name="name" type="string">
      Nome da ferramenta (minúsculas com underscore)
    </ResponseField>

    <ResponseField name="description" type="string">
      Explicação detalhada de quando e como a IA deve utilizar esta ferramenta
    </ResponseField>

    <ResponseField name="endpoint" type="string">
      URL do endpoint da API que será chamado
    </ResponseField>

    <ResponseField name="method" type="string">
      Método HTTP (GET, POST, PUT, PATCH, DELETE)
    </ResponseField>

    <ResponseField name="timeout" type="integer">
      Timeout da requisição em segundos (1-30)
    </ResponseField>

    <ResponseField name="headers" type="array">
      Headers HTTP enviados na requisição
      <Expandable title="propriedades de headers">
        <ResponseField name="name" type="string">
          Nome do header
        </ResponseField>
        <ResponseField name="value" type="string">
          Valor do header
        </ResponseField>
      </Expandable>
    </ResponseField>

    <ResponseField name="schema" type="array">
      Parâmetros que a IA irá extrair e enviar para o endpoint
      <Expandable title="propriedades de schema">
        <ResponseField name="name" type="string">
          Nome do parâmetro
        </ResponseField>

        <ResponseField name="type" type="string">
          Tipo do parâmetro (string, number, boolean)
        </ResponseField>

        <ResponseField name="description" type="string">
          Descrição para ajudar a IA a entender como extrair este parâmetro
        </ResponseField>
      </Expandable>
    </ResponseField>

    <ResponseField name="created_at" type="string">
      Timestamp ISO 8601 de criação da ferramenta
    </ResponseField>

    <ResponseField name="updated_at" type="string">
      Timestamp ISO 8601 da última atualização da ferramenta
    </ResponseField>
  </Expandable>
</ResponseField>
